Tipos de Dados Básicos

  • int: Números inteiros (10, -5, 0)
  • float: Números decimais (3.14, -0.001)
  • str: Texto ("Olá", 'Python')
  • bool: Valores lógicos (True, False)
  • None: Representa ausência de valor

Verificação de Tipos

# Verificar o tipo de uma variável

Loop while

# Executa enquanto a condição for verdadeira
contador = 0
while contador < 5:
    print(contador)
    contador += 1

# break e continue
while True:
    resposta = input("Digite 'sair' para encerrar: ")
    if resposta == "sair":
        break  # Sai do loop
    if resposta == "":
        continue  # Volta ao início do loop

Loop for

# Iterando sobre uma lista
frutas = ["maçã", "banana", "laranja"]
for fruta in frutas:
    print(fruta)

# Iterando com índices
for i in range(len(frutas)):
    print(f"{i}: {frutas[i]}")

# Função range()
for i in range(5):  # 0, 1, 2, 3, 4
    print(i)

for i in range(2, 8):  # 2, 3, 4, 5, 6, 7
    print(i)

for i in range(1, 10, 2):  # 1, 3, 5, 7, 9
    print(i)

# Iterando sobre dicionários
pessoa = {"nome": "Ana", "idade": 28}
for chave in pessoa:
    print(f"{chave}: {pessoa[chave]}")

# Desempacotando itens
for chave, valor in pessoa.items():
    print(f"{chave}: {valor}")

# Enumerate para obter índice e valor
for i, fruta in enumerate(frutas):
    print(f"{i}: {fruta}")

Manipulação de Erros e Exceções

Python Automação

Estrutura try-except

# Tratamento básico de exceções
try:
    numero = int(input("Digite um número: "))
    resultado = 10 / numero
    print(f"Resultado: {resultado}")
except ValueError:
    print("Entrada inválida. Digite um número.")
except ZeroDivisionError:
    print("Não é possível dividir por zero.")
except:
    print("Ocorreu um erro inesperado.")

# Capturando a exceção
try:
    # Código que pode gerar erro
    arquivo = open("dados.txt", "r")
except Exception as e:
    print(f"Erro: {e}")

# Bloco else (executado se não houver exceção)
try:
    dados = [1, 2, 3]
    valor = dados[1]
except IndexError:
    print("Índice inválido")
else:
    print(f"Valor obtido: {valor}")

Bloco finally

# Executado sempre, independente de exceção
try:
    arquivo = open("log.txt", "w")
    arquivo.write("Teste")
except IOError:
    print("Erro ao manipular o arquivo")
finally:
    # Garantir que o arquivo seja fechado
    arquivo.close()

Gerenciador de contexto (with)

# Forma mais segura de trabalhar com recursos
with open("dados.txt", "r") as arquivo:
    conteudo = arquivo.read()
# Arquivo é fechado automaticamente

Exceções comuns em automação

  • FileNotFoundError: Arquivo não encontrado
  • PermissionError: Sem permissão para acessar
  • TimeoutError: Operação excedeu o tempo limite
  • ConnectionError: Erro de conexão
  • KeyError: Chave não encontrada em dicionário
  • IndexError: Índice fora dos limites

Aplicando os conceitos aprendidos

# Script para processar arquivos de dados
import os
import csv
from datetime import datetime

def listar_arquivos(diretorio, extensao=".csv"):
    """
    Lista todos os arquivos com a extensão especificada em um diretório.
    
    Args:
        diretorio (str): Caminho do diretório a ser pesquisado
        extensao (str): Extensão dos arquivos a serem listados
        
    Returns:
        list: Lista de caminhos completos dos arquivos encontrados
    """
    arquivos = []
    try:
        for arquivo in os.listdir(diretorio):
            if arquivo.endswith(extensao):
                caminho_completo = os.path.join(diretorio, arquivo)
                arquivos.append(caminho_completo)
        return arquivos
    except FileNotFoundError:
        print(f"Diretório não encontrado: {diretorio}")
        return []

def processar_arquivo(caminho_arquivo):
    """
    Processa um arquivo CSV e retorna estatísticas básicas.
    
    Args:
        caminho_arquivo (str): Caminho do arquivo a ser processado
        
    Returns:
        dict: Dicionário com estatísticas do arquivo
    """
    try:
        with open(caminho_arquivo, 'r', encoding='utf-8') as arquivo:
            leitor = csv.reader(arquivo)
            cabecalho = next(leitor)  # Pula o cabeçalho
            
            linhas = list(leitor)
            total_linhas = len(linhas)
            
            # Calcular estatísticas (exemplo simples)
            estatisticas = {
                "arquivo": os.path.basename(caminho_arquivo),
                "total_linhas": total_linhas,
                "colunas": cabecalho,
                "processado_em": datetime.now().strftime("%Y-%m-%d %H:%M:%S")
            }
            
            return estatisticas
    except Exception as e:
        print(f"Erro ao processar {caminho_arquivo}: {e}")
        return None

def main():
    # Configurações
    DIRETORIO_DADOS = "./dados"
    
    # Listar arquivos
    arquivos = listar_arquivos(DIRETORIO_DADOS)
    
    if not arquivos:
        print("Nenhum arquivo CSV encontrado.")
        return
    
    # Processar cada arquivo
    resultados = []
    for arquivo in arquivos:
        print(f"Processando {os.path.basename(arquivo)}...")
        estatisticas = processar_arquivo(arquivo)
        if estatisticas:
            resultados.append(estatisticas)
    
    # Exibir resultados
    print("\nResumo do processamento:")
    for resultado in resultados:
        print(f"Arquivo: {resultado['arquivo']}")
        print(f"  Linhas: {resultado['total_linhas']}")
        print(f"  Colunas: {', '.join(resultado['colunas'])}")
        print(f"  Processado em: {resultado['processado_em']}")
        print("-" * 40)
    
    print(f"Total de arquivos processados: {len(resultados)}")

if __name__ == "__main__":
    main()
